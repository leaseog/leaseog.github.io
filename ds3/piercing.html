back to main // Get access to the camera async function initCamera() { try { const constraints = { video: true, // Enable video audio: false // Disable audio }; const stream = await navigator.mediaDevices.getUserMedia(constraints); const videoElement = document.getElementById('video'); videoElement.srcObject = stream; // Set the video source to the camera stream } catch (error) { console.error('Error accessing the camera: ', error); } } // Initialize the camera when the page loads window.onload = initCamera;
